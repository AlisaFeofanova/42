########### Successful Scripts (Comparison with Bash) ###############

./pipex infile.txt "cat" "wc -l" outfile_pipex.txt
< infile.txt cat | wc -l > outfile_bash.txt
diff outfile_pipex.txt outfile_bash.txt && echo "✅ TEST 1.1 PASSED"

./pipex infile.txt "grep 42" "wc -w" outfile_pipex.txt
< infile.txt grep 42 | wc -w > outfile_bash.txt
diff outfile_pipex.txt outfile_bash.txt && echo "✅ TEST 1.2 PASSED"

./pipex infile.txt "/bin/cat" "/usr/bin/wc -c" outfile_pipex.txt
< infile.txt /bin/cat | /usr/bin/wc -c > outfile_bash.txt
diff outfile_pipex.txt outfile_bash.txt && echo "✅ TEST 1.3 PASSED"
